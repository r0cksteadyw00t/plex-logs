function CleanJson {
    param([string]$Text)
    if ($null -eq $Text) { return "" }
    $S = [string]$Text
    $S = $S.Trim()
    if ($S.Length -gt 0 -and [int][char]$S.Substring(0,1) -eq 65279) { $S = $S.Substring(1) }
    $F = $S.IndexOf("{")
    $L = $S.LastIndexOf("}")
    if ($F -ge 0 -and $L -gt $F) { $S = $S.Substring($F, ($L - $F + 1)) }
    return $S.Trim()
}

    param([string]$Name)

}
